on_game_start = {
	effect = {
		# 批量按地区设置科技等级
		every_country = {
			set_starting_tech_level_by_region = yes
			# 根据科技等级给予思潮
			give_institutions_by_tech_level = yes
            #设置自动征服 - 给所有首都在中国的国家
            if = {
                limit = {
                    exists = capital
                    OR = {
                        capital.region = region:west_china_region
                        capital.region = region:east_china_region
                        capital.region = region:north_china_region
                        capital.region = region:south_china_region
                        capital.region = region:manchuria_region #满洲
                    }
                }
                add_country_modifier = { modifier = 1644_china_auto_conquest years = 8 }
            }

            # 开局生成大量女角色以防止绝嗣（为每个国家触发）
            trigger_event_silently = zzz_1644_generate_female_characters.1
        }
        # 清除默认佣兵队长（仅针对主要国家）
        c:QNG = { trigger_event_silently = cleanup_1644.1 }
        c:CSH = { trigger_event_silently = cleanup_1644.1 }
        c:WSG = { trigger_event_silently = cleanup_1644.1 }
        c:CXI = { trigger_event_silently = cleanup_1644.1 }
        c:MNG = { trigger_event_silently = cleanup_1644.1 }

        # 主要势力开局事件
        c:CSH = {
            trigger_event_silently = csh_1644_start_army.1
            trigger_event_silently = { id = csh_1644_unification.1 days = 4 }
            unlock_government_reform_effect = { type = chi_peasant_empire_reform }
        }
        c:QNG = {
            trigger_event_silently = qng_1644_banner_army.1
            trigger_event_silently = { id = qng_1644_unification.1 days = 15 }
        }
        c:WSG = {
            trigger_event_silently = wsg_1644_start_army.1
            c:WSG = {
                if = {
                    limit = { NOT = { has_variable = wsg_goto_beijing } }
                    trigger_event_silently = { id = wsg_1644_choice.2 days = 1 }
                }
            }
        }
        c:MNG = { trigger_event_silently = mng_1644.1 }
        c:CXI = { trigger_event_silently = cxi_1644_start_army.1 }
        
        # 开局生成大量女角色以防止绝嗣
   
        # 开局生成明末军阀军队（江北四镇、左良玉、大同镇）
        c:CGJ = { trigger_event_silently = ming_warlords.1 } # 兴平镇（高杰）
        c:LZQ = { trigger_event_silently = ming_warlords.1 } # 东平镇（刘泽清）
        c:LLZ = { trigger_event_silently = ming_warlords.1 } # 广昌镇（刘良佐）
        c:HDG = { trigger_event_silently = ming_warlords.1 } # 靖南镇（黄得功）
        c:ZLY = { trigger_event_silently = ming_warlords.1 } # 宁南镇（左良玉）
        c:CJX = { trigger_event_silently = ming_warlords.1 } # 大同镇（姜瓖）

    }
}
